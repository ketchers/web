def GS(B, ip):
    if len(B) == 1:
        b = B[0]
        return [b/sqrt(ip(b,b))]
    else:
        C = GS(B[:-1], ip)
        b = B[-1] - proj(C, ip, B[-1])
        if ip(b,b) != 0:
            return C + [b/sqrt(ip(b,b))]
        return C
